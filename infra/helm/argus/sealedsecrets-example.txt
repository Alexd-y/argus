# ──────────────────────────────────────────────────────────────────────────────
# ARGUS — SealedSecret EXAMPLES (NOT applied by Helm)
# ──────────────────────────────────────────────────────────────────────────────
# This file lives at the chart ROOT (NOT under `templates/`) so Helm's loader
# never tries to render it. The `.txt` extension is also accepted by `helm lint`
# whitelist. The file is intentionally non-actionable — it only documents the
# `kubeseal` workflow and lists the secrets the chart's deployments expect.
#
# SealedSecrets MUST be encrypted against your cluster's specific public key —
# generated by `kubeseal` on the operator workstation. Plaintext Secret
# manifests MUST NEVER be committed to Git (see the CI guard
# `scripts/check_no_plaintext_secrets.py`).
#
# ─────────────────────────────────────────────────────────────────────────────
# Operator runbook (mirrored in docs/deployment-helm.md)
# ─────────────────────────────────────────────────────────────────────────────
#
#   1. Drop a plain Secret into a *temporary* file (NEVER commit):
#
#        cat <<EOF > /tmp/argus-postgres.yaml
#        apiVersion: v1
#        kind: Secret
#        metadata:
#          name: argus-postgres
#          namespace: argus-prod
#        type: Opaque
#        stringData:
#          password: "<rotate-me>"
#        EOF
#
#   2. Seal it against the cluster:
#
#        kubeseal --controller-namespace=kube-system \
#                 --controller-name=sealed-secrets-controller \
#                 --format=yaml \
#                 < /tmp/argus-postgres.yaml \
#                 > infra/secrets/argus-postgres-sealed.yaml
#
#   3. Commit the sealed manifest (NEVER the plaintext) and `kubectl apply`:
#
#        git add infra/secrets/argus-postgres-sealed.yaml
#        kubectl apply -f infra/secrets/argus-postgres-sealed.yaml
#
#   4. Wipe the plaintext (PowerShell or POSIX):
#
#        Remove-Item /tmp/argus-postgres.yaml      # PowerShell
#        shred -u   /tmp/argus-postgres.yaml       # POSIX
#
# ─────────────────────────────────────────────────────────────────────────────
# What the sealed object looks like once `kubeseal` finishes
# ─────────────────────────────────────────────────────────────────────────────
#
# apiVersion: bitnami.com/v1alpha1
# kind: SealedSecret
# metadata:
#   name: argus-postgres
#   namespace: argus-prod
# spec:
#   encryptedData:
#     password: AgB...REDACTED...
#   template:
#     metadata:
#       name: argus-postgres
#       namespace: argus-prod
#     type: Opaque
#
# ─────────────────────────────────────────────────────────────────────────────
# Required SealedSecrets for an ARGUS production deployment
# ─────────────────────────────────────────────────────────────────────────────
#
#   * argus-postgres   — keys: password
#   * argus-redis      — keys: password
#   * argus-minio      — keys: rootUser, rootPassword
#   * argus-llm        — keys: openai_api_key, anthropic_api_key
#   * argus-webhooks   — keys: slack_signing_secret              (optional)
#   * argus-tls        — TLS cert/key pair if NOT using cert-manager
#
# Each name above MUST match the `existingSecretName` keys under `secrets.*`
# in `values.yaml` (and any per-environment override). The deployment
# templates resolve them via `valueFrom.secretKeyRef`, so a missing or
# mistyped name will surface as a `CreateContainerConfigError` on the first
# pod start.
#
# ─────────────────────────────────────────────────────────────────────────────
# CI guard
# ─────────────────────────────────────────────────────────────────────────────
#
# `scripts/check_no_plaintext_secrets.py` refuses to merge any `kind: Secret`
# manifest under `infra/` outside of fixtures and tests. The same script also
# flags any `Secret`-shaped Helm template that hard-codes `data:` /
# `stringData:` payloads, ensuring the chart never bypasses the SealedSecrets
# pipeline.
